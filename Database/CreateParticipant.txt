CREATE TABLE participant (
username			VARCHAR(30),
meetingID			INT,
attendence			INT,
viability 			INT,
alarmtid			DATETIME,
answerd				BOOLEAN,
participantChange	BOOLEAN,
timeChange			BOOLEAN,
placeChange			BOOLEAN,
descriptionChange	BOOLEAN,	


FOREIGN KEY (meetingID) REFERENCES meeting(meetingID) 	ON DELETE CASCADE
														ON UPDATE CASCADE,
FOREIGN KEY (username) REFERENCES users(username) 		ON DELETE CASCADE
														ON UPDATE CASCADE
);
