create table meeting(
	
meetingID		INT AUTO_INCREMENT,
mDescription 	VARCHAR(140),
timeStart		DATETIME,
timeEnd			DATETIME,
sted			VARCHAR(30),
nOfParticipant	INT,
roomID			REFERENCES room(roomID) 	ON DELETE CASCADE
											ON UPDATE CASCADE,	
username		REFERENCES users(username) 	ON DELETE CASCADE
											ON UPDATE CASCADE,	
PRIMARY KEY 		(meetingID));
